1. find the smaller angle

        var Minimal_Angle = (h, m) => 
{
    h=h%12;
    let min=(m*360)/60;
    let hour=(h*360)/12+(m*360)/(12*60);
    let angle=Math.abs(hour-min);
    if(angle>180){
      return 360-angle;
    }
    else{
      return angle;
    }
};

var Reverse_Number = (N) => 
        {
          let b, sum=0
          while(N>0){
          b = N % 10;
              sum = sum * 10 + b;
              N = parseInt(N / 10);
          }
         return sum;
        };
